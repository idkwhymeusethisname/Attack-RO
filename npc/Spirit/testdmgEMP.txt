mooc,227,66,0	script	Emperium#2	844,{
	end;
	
OnMyMobDead:
OnInit:
	getunitdata(getnpcid(0), .@d);
	monster strnpcinfo(4),.@d[UNPC_X],.@d[UNPC_Y],"Emperium",3998,1,strnpcinfo(0)+"::OnMyMobDead";
	.@dupid = atoi(strnpcinfo(2));
	.gid[.@dupid] = $@mobid[0];
	.mhp[.@dupid] = strmobinfo(4,3998);
	setunitdata .gid[.@dupid],UMOB_MODE,.@d[UMOB_MODE]|0x6370000|0x0200000; // Set MD_STATUS_IMMUNE (0x6370000) because EDP/burn/%MHP based status can skew the DPS counter so much.
	initnpctimer;
	end;
	
OnTimer5000:
	.@dupid = atoi(strnpcinfo(2));
	if(unitexists(.gid[.@dupid])){
		getunitdata(.gid[.@dupid],.@d);
		.@diff = (.mhp[.@dupid] - .@d[UMOB_HP]);
		if(.@diff > 0){
//			npctalk "ทดสอบดาเมจ : ฉันได้รับ " + (.@diff/5) + " ดาเมจ เมื่อ 5 วินาทีที่แล้ว.";
//			setunitdata .gid[.@dupid],UMOB_HP,.mhp[.@dupid];
//			specialeffect EF_HEAL;
		}
	}
	initnpctimer;
}
